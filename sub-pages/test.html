function getImages() {
    let imageArray = document.querySelectorAll(".gallery-item");

    let imgSrcArray = [];

    imageArray.forEach((i) => {
        let image = i.querySelector("img").getAttribute("src");
        imgSrcArray.push(image);
    });
    return imgSrcArray;
}

function addJSzip() {
    return new Promise((resolve) => {
        let script = document.createElement("script");
        script.src =
            "https://cdnjs.cloudflare.com/ajax/libs/jszip/3.10.1/jszip.min.js";
        script.onload = resolve;
        document.head.appendChild(script);
    });
}

function getCleanTitle() {
    let h1 = document.querySelector("h1.entry-title");
    if (!h1) return "Galley";
    let title = h1.textContent.trim();
    title = title.replace(/["“”'].*?["“”']/g, "");
    title = title.replace(/\b\d+\s+(photos?|videos?)\b/gi, "");
    title = title.replace(/[^\x00-\x7F]/g, "");
    title = title.replace(/[^a-zA-Z0-9\- ]/g, "");
    title = title.replace(/\s+/g, " ").trim();

    return title;
}

async function downloadAndZip(url_list, filename) {
    let zip = new JSZip();
    let imagesPromises = url_list.map(async (url, index) => {
        let response = await fetch(url);
        let blob = await response.blob();

        let reverseIndex = url_list.length - index;
        zip.file(`Photo${reverseIndex}.jpg`, blob);
    });
    await Promise.all(imagesPromises);

    zip.generateAsync({ type: "blob" }).then(function (content) {
        let link = document.createElement("a");
        link.href = URL.createObjectURL(content);
        link.download = `${filename}.zip`;
        link.click();
    });
}

async function main() {
    await addJSzip();
    let title = getCleanTitle();
    let imageUrl = getImages();
    await downloadAndZip(imageUrl, title);
}
